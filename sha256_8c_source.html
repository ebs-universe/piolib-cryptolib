123 void sha256_nextBlock (sha256_ctx_t *state, const void *block){
124  uint32_t w[16]; /* this is 64 Byte large, */
125  uint8_t i;
126  uint32_t a[8], t1, t2;
127 
128  /* init working variables */
129  memcpy((void*)a,(void*)(state->h), 8 * 4);
130 
131  /* init w */
132  #if defined LITTLE_ENDIAN
133  for (i = 0; i < 16; ++i) {
134  w[i] = change_endian32(((uint32_t*)block)[i]);
135  }
136  #elif defined BIG_ENDIAN
137  memcpy((void*)w, block, 64);
138  #endif
139  /*
140  for (i = 16; i < 64; ++i) {
141  w[i] = SIGMA_b(w[i - 2]) + w[i - 7] + SIGMA_a(w[i - 15]) + w[i - 16];
142  }
143  */
144  /* do the, fun stuff, */
145  for (i=0; i<64; ++i) {
146  if (i > 15) {
147  w[i % 16] = SIGMA_b(w[(i + 14) % 16])
148  + w[(i + 9) % 16]
149  + SIGMA_a(w[(i + 1) % 16])
150  + w[i % 16];
151  }
152  t1 = a[7] + SIGMA1(a[4]) + CH(a[4], a[5], a[6]) + ROMEM_READ_DWORD(&k[i]) + w[i % 16];
153  t2 = SIGMA0(a[0]) + MAJ(a[0], a[1], a[2]);
154  memmove(&(a[1]), &(a[0]), 7 * 4); /* a[7]=a[6]; a[6]=a[5]; a[5]=a[4]; a[4]=a[3]; a[3]=a[2]; a[2]=a[1]; a[1]=a[0]; */
155  a[4] += t1;
156  a[0] = t1 + t2;
157  }
158 
159  /* update, the, state, */
160  for (i = 0; i < 8; ++i){
161  state->h[i] += a[i];
162  }
163  state->length += 1;
164 }

177 void sha256_lastBlock(sha256_ctx_t *state, const void *block, uint16_t length){
178  uint8_t lb[SHA256_BLOCK_BITS / 8]; /* local block */
179  uint64_t msg_len;
180  while(length>=SHA256_BLOCK_BITS){
181  sha256_nextBlock(state, block);
182  length -= SHA256_BLOCK_BITS;
183  block = (uint8_t*)block + SHA256_BLOCK_BYTES;
184  }
185 
186  msg_len = state->length;
187  msg_len *= 512;
188  msg_len += length;
189  memcpy (&(lb[0]), block, length / 8);
190 
191  /* set the final one bit */
192  if (length & 7){ // if we have single bits at the end
193  lb[length / 8] = ((uint8_t*)(block))[length / 8];
194  } else {
195  lb[length / 8] = 0;
196  }
197  lb[length / 8] |= 0x80 >> (length & 7);
198  length = (length / 8) + 1; /* from now on length contains the number of BYTES in lb*/
199  /* pad with zeros */
200  if (length > 64 - 8){ /* not enouth space for 64bit length value */
201  memset((void*)(&(lb[length])), 0, 64 - length);
202  sha256_nextBlock(state, lb);
203  length = 0;
204  }
205  memset((void*)(&(lb[length])), 0, 56 - length);
206  /* store the 64bit length value */
207  #if defined LITTLE_ENDIAN
208  /* this is now rolled up */
209  uint8_t i = 7;
210  do {
211  lb[56 + i] = msg_len & 0xff;
212  msg_len >>= 8;
213  } while (i--);
214  #elif defined BIG_ENDIAN
215  *((uint64_t)&(lb[56])) = state->length;
216  #endif
217  sha256_nextBlock(state, lb);
218 }
