86 void md5_core(uint32_t *a, void *block, uint8_t as, uint8_t s, uint8_t i, uint8_t fi){
87  uint32_t t;
88  md5_func_t *funcs[]={md5_F, md5_G, md5_H, md5_I};
89  as &= 0x3;
90  /* a = b + ((a + F(b,c,d) + X[k] + T[i]) <<< s). */
91  #ifdef DEBUG
92  char funcc[]={'*', '-', '+', '~'};
93  cli_putstr("\r\n DBG: md5_core [");
94  cli_putc(funcc[fi]);
95  cli_hexdump(&as, 1); cli_putc(' ');
96  cli_hexdump(&k, 1); cli_putc(' ');
97  cli_hexdump(&s, 1); cli_putc(' ');
98  cli_hexdump(&i, 1); cli_putc(']');
99  #endif
100  t = a[as] + funcs[fi](a[(as+1)&3], a[(as+2)&3], a[(as+3)&3])
101  + *((uint32_t*)block) + ROMEM_READ_DWORD(md5_T_p+i) ;
102  a[as]=a[(as+1)&3] + ROTL32(t, s);
103 }

105 void md5_nextBlock(md5_ctx_t *state, const void *block){
106  uint32_t a[4];
107  uint8_t m,n,i=0;
108  /* this requires other mixed sboxes */
109  #ifdef DEBUG
110  cli_putstr("\r\n DBG: md5_nextBlock: block:\r\n");
111  cli_hexdump(block, 16); cli_putstr("\r\n");
112  cli_hexdump(block+16, 16); cli_putstr("\r\n");
113  cli_hexdump(block+32, 16); cli_putstr("\r\n");
114  cli_hexdump(block+48, 16); cli_putstr("\r\n");
115  #endif
116 
117  a[0]=state->a[0];
118  a[1]=state->a[1];
119  a[2]=state->a[2];
120  a[3]=state->a[3];
121 
122  /* round 1 */
123  uint8_t s1t[]={7,12,17,22}; // 1,-1 1,4 2,-1 3,-2
124  for(m=0;m<4;++m){
125  for(n=0;n<4;++n){
126  md5_core(a, &(((uint32_t*)block)[m*4+n]), 4-n, s1t[n],i++,0);
127  }
128  }
129  /* round 2 */
130  uint8_t s2t[]={5,9,14,20}; // 1,-3 1,1 2,-2 2,4
131  for(m=0;m<4;++m){
132  for(n=0;n<4;++n){
133  md5_core(a, &(((uint32_t*)block)[(1+m*4+n*5)&0xf]), 4-n, s2t[n],i++,1);
134  }
135  }
136  /* round 3 */
137  uint8_t s3t[]={4,11,16,23}; // 0,4 1,3 2,0 3,-1
138  for(m=0;m<4;++m){
139  for(n=0;n<4;++n){
140  md5_core(a, &(((uint32_t*)block)[(5-m*4+n*3)&0xf]), 4-n, s3t[n],i++,2);
141  }
142  }
143  /* round 4 */
144  uint8_t s4t[]={6,10,15,21}; // 1,-2 1,2 2,-1 3,-3
145  for(m=0;m<4;++m){
146  for(n=0;n<4;++n){
147  md5_core(a, &(((uint32_t*)block)[(0-m*4+n*7)&0xf]), 4-n, s4t[n],i++,3);
148  }
149  }
150  state->a[0] += a[0];
151  state->a[1] += a[1];
152  state->a[2] += a[2];
153  state->a[3] += a[3];
154  state->counter++;
155 }

157 void md5_lastBlock(md5_ctx_t *state, const void *block, uint16_t length_b){
158  uint16_t l;
159  union {
160  uint8_t v8[64];
161  uint64_t v64[ 8];
162  } buffer;
163  while (length_b >= 512){
164  md5_nextBlock(state, block);
165  length_b -= 512;
166  block = ((uint8_t*)block) + 512/8;
167  }
168  memset(buffer.v8, 0, 64);
169  memcpy(buffer.v8, block, length_b/8);
170  /* insert padding one */
171  l=length_b/8;
172  if(length_b%8){
173  uint8_t t;
174  t = ((uint8_t*)block)[l];
175  t |= (0x80>>(length_b%8));
176  buffer.v8[l]=t;
177  }else{
178  buffer.v8[l]=0x80;
179  }
180  /* insert length value */
181  if(l+sizeof(uint64_t) >= 512/8){
182  md5_nextBlock(state, buffer.v8);
183  state->counter--;
184  memset(buffer.v8, 0, 64-8);
185  }
186  buffer.v64[7] = (state->counter * 512) + length_b;
187  md5_nextBlock(state, buffer.v8);
188 }

194 void md5(md5_hash_t *dest, const void *msg, uint32_t length_b){
195  md5_ctx_t ctx;
196  md5_init(&ctx);
197  while(length_b>=MD5_BLOCK_BITS){
198  md5_nextBlock(&ctx, msg);
199  msg = (uint8_t*)msg + MD5_BLOCK_BYTES;
200  length_b -= MD5_BLOCK_BITS;
201  }
202  md5_lastBlock(&ctx, msg, length_b);
203  md5_ctx2hash(dest, &ctx);
204 }
